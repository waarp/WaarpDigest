46  class MD5State {
47      /**
48       * 128-bit state
49       */
50      int state[];
51  
52      /**
53       * 64-bit character count
54       */
55      long count;
56  
57      /**
58       * 64-byte buffer (512 bits) for storing to-be-hashed characters
59       */
60      byte buffer[];
61  
62      /**
63       * Constructor
64       * 
65       */
66      MD5State() {
67          buffer = new byte[64];
68          count = 0;
69          state = new int[4];
70  
71          state[0] = 0x67452301;
72          state[1] = 0xefcdab89;
73          state[2] = 0x98badcfe;
74          state[3] = 0x10325476;
75  
76      }
77  
78      /**
79       * Create this State as a copy of another state
80       * 
81       * @param from
82       */
83      MD5State(MD5State from) {
84          this();
85  
86          int i;
87  
88          for (i = 0; i < buffer.length; i++) {
89              buffer[i] = from.buffer[i];
90          }
91  
92          for (i = 0; i < state.length; i++) {
93              state[i] = from.state[i];
94          }
95  
96          count = from.count;
97      }
98  }
